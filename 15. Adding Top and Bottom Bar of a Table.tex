		\hline
		Set Theory & 2.00 & 25 \\
		\hline
		Linear Algebra & 3.00 & 57 \\
		\hline
		Quantum Computing & 3.00 & 108 \\
		\hline
		% it will add a horizontal bar of size 2 pt at the bottom of the table
		\bottomrule[2 pt]
	\end{tabular}
\end{document}
